# [200. 岛屿数量](https://leetcode.cn/problems/number-of-islands/)

```java
class Solution {
    public int numIslands(char[][] grid) {
        int res = 0;
        int m = grid.length;
        int n = grid[0].length;
        for(int i = 0;i<m;i++){
            for(int j = 0;j<n;j++){
                if(grid[i][j] == '1'){
                    burn(i,j,grid);
                    res++;
                }    
            }
        }
        return res;
    }

    public void burn(int i, int j, char[][] grid){
        
        if(i<0||i>=grid.length||j<0||j>=grid[0].length){
            return;
        }
        if(grid[i][j]!='1'){
            return;
        }
        grid[i][j] = '2';
        
        burn(i-1,j,grid);
        burn(i+1,j,grid);
        burn(i,j-1,grid);
        burn(i,j+1,grid);
    }
}
```
